```python
# 集合无序，唯一的
# 无序：不可用下标来取值
# 唯一：可以去重+-

#创建集合

#set()方式创建 ，里面可以是列表，元组，字典，字符串
set1=set([1,2,3,4,5,6])
print(set1) #{1, 2, 3, 4, 5, 6}
set1=set((1,2,3,4,5,6))
print(set1) #{1, 2, 3, 4, 5, 6}
set1=set({1:2,3:4,5:6})
print(set1) #{1,  3,  5}
set1=set('12345')
print(set1) #{'2', '4', '1', '3', '5'}

# {}方式创建，里面可以是数字，字符串，元组

set1 ={1,2,3,'123',(1,2,3)}
print(set1)

#set1 ={1,2,3,'123',[1,2,3]}   {}方式放集合会报错

```
